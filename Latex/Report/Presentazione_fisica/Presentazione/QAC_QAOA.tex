\section{QAC to QAOA}
\begin{frame}{QAOA}
	\begin{block}{Quantum approximate optimization algorithm}
		\begin{enumerate}
			\item Definire matrici: $\mathcal{H}_c$ e  $\ \mathcal{H}_m$
			\item Definire oracoli parametrici: $\mathcal{U}_c(\gamma) = \mathnormal{e}^{-i\gamma\mathcal{H}_c}$ e $\ \mathcal{U}_m(\beta) = \mathnormal{e}^{-i\beta\mathcal{H}_m}$
			\item Applicazione ripetuta di $\mathcal{U}_c(\gamma)$ e $\ \mathcal{U}_m(\gamma)$
			\item Ottimizzazione (classica) dei parametri $\gamma_i$ e $\beta_i$
		\end{enumerate}
	\end{block}
	\begin{figure}[h]
		\includegraphics[width=.63\linewidth]{QAOA}
		\caption{BLEKOS, Kostas, et al. A review on quantum approximate optimization algorithm and its variants}
	\end{figure}
\end{frame}
