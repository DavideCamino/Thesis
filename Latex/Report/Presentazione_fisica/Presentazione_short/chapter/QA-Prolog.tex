\begin{frame}[fragile]{QA-Prolog}
	\begin{minipage}{.42\linewidth}

			\begin{itemize}
				\item Traduzione da Prolog a Verilog
				\item Wrapper per tutta la Pipeline
				\item Risultati formato Human Readable
				\item Dimensionamento automatico variabili
			\end{itemize}
		\end{block}

		\begin{figure}[h]

sat(A, B, C, Y) :-

	not(X, Z),
	and(A, Z, Y).
				\end{minted}
				\hrule
			\end{minipage}
			\caption{Prolog Code}
		\end{figure}
	\end{minipage}
